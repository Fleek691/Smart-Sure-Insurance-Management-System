# .env loader for .NET
# Add this NuGet package to your project:
#   DotNetEnv (https://www.nuget.org/packages/DotNetEnv)
#
# In Program.cs, add this at the top before building the app:
#   DotNetEnv.Env.Load();
#
# This will load variables from a .env file into the environment.
#
# In appsettings.json, you can reference env variables using:
#   "Jwt:Key": "${JWT_SECRET}"
#
# Or, in code, use:
#   Environment.GetEnvironmentVariable("JWT_SECRET")
#
# For production, set these variables in your deployment environment.
